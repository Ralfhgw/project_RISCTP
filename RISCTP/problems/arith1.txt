theorem T5 ⇔ ∀x:Int,y:Int. ¬(x = y) ⇒ x < y ∨ x > y;
theorem T6 ⇔ ∀a:Int,b:Int. a = b ⇒ ¬(a < b);
theorem T7 ⇔ ∀x:Int,y:Int. x ≠ y ⇒ x < y ∨ x > y;
